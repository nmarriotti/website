Template classes allow your class variable data types to differ from object to object if necessary. In this tutorial we'll create a class that requires two variables be sent to the constructor.

Here is the template we'll be using:


template <typename TYPE1=int, typename TYPE2=double> /* Default is int, double */

Here is the class we'll use:


class Example {
    private:
        TYPE1 value1;
        TYPE2 value2;

    public:
        Example(TYPE1 val1, TYPE2 val2) {
            cout << "Constructor" << endl;
            value1 = val1;
            value2 = val2;
        }

        TYPE1 get_value1() {
            return value1;
        }

        TYPE2 get_value2() {
            return value2;
        }
};

When we created our templete we set our TYPE1 and TYPE2 to our default data types of int and double. When we create an object using our class we can change these data types if needed.

Creating objects

Now it's time to use our template and class to create some objects. In the first example we'll use the default data types. In the second example we'll pass in a string to the constructor.


    Example<> object1 (29, 49.99); /* use default data types */
    cout << object1.get_value1() << endl;
    cout << object1.get_value2() << endl;

    Example<int, string> object2 (29, "Learning templates in C++"); /* use int, string data type */
    cout << object2.get_value1() << endl;
    cout << object2.get_value2() << endl;

Complete code

Here is the complete source code for this tutorial.


#include <iostream>

using namespace std;

/*
Class templates allow you to pick which data type is used for variables.
*/

template <typename TYPE1=int, typename TYPE2=double> /* Default is int, double */
class Example {
    private:
        TYPE1 value1;
        TYPE2 value2;

    public:
        Example(TYPE1 val1, TYPE2 val2) {
            cout << "Constructor" << endl;
            value1 = val1;
            value2 = val2;
        }

        TYPE1 get_value1() {
            return value1;
        }

        TYPE2 get_value2() {
            return value2;
        }
};

int main()
{
    Example<> object1 (29, 49.99); /* use default data types */
    cout << object1.get_value1() << endl;
    cout << object1.get_value2() << endl;

    Example<int, string> object2 (29, "Learning templates in C++"); /* use int, string data type */
    cout << object2.get_value1() << endl;
    cout << object2.get_value2() << endl;


    return 0;
}
